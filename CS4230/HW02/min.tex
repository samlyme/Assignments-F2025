    \item {
        Density (3 points)
        \begin{itemize}
            \item {
                Write the formula for density.

                \[D = \frac{2E}{N(N-1)}\]
            }

            \item {
                Count nodes and edges.

                8 nodes, 10 edges.
            }

    \item {

                \[C_i = \frac{2E_i}{k_i(k_i-1)}\]
            }
